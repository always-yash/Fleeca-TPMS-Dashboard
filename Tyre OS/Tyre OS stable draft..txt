    <script>
        document.addEventListener("DOMContentLoaded", () => {
            const sectorBgs = document.querySelectorAll(".wheel-sector-btn");
            const overlayBlocks = document.querySelectorAll(".sector-content-block");
            const centerHubBtn = document.getElementById("center-hub-btn");

            // Unified State Processor Engine
            function executeToggle(sectorId, forceState = null) {
                const targetWedge = document.getElementById(`btn-${sectorId}`);
                const targetCard = document.getElementById(`card-${sectorId}`);
                const targetNode = document.getElementById(`node-${sectorId}`);
                const targetOverlay = document.getElementById(`block-${sectorId}`);

                // Determine target binary outcome
                const finalState = (forceState !== null) ? forceState : !targetWedge.classList.contains("is-active");

                if (finalState) {
                    targetWedge.classList.add("is-active");
                    if (targetCard) targetCard.classList.add("is-active");
                    if (targetNode) targetNode.classList.add("is-active");
                    if (targetOverlay) targetOverlay.classList.add("is-active");
                } else {
                    targetWedge.classList.remove("is-active");
                    if (targetCard) targetCard.classList.remove("is-active");
                    if (targetNode) targetNode.classList.remove("is-active");
                    if (targetOverlay) targetOverlay.classList.remove("is-active");
                }

                // DYNAMIC ASSISTING ICON STATE SWAPPER
                if (targetOverlay) {
                    const iconImg = targetOverlay.querySelector(".assisting-icon-img");
                    if (iconImg) {
                        let currentSrc = iconImg.getAttribute("src");
                        if (finalState) {
                            // Turn to normal white icon when active
                            currentSrc = currentSrc.replace("-orange.png", ".png");
                        } else {
                            // Revert to orange background icon when inactive
                            if (!currentSrc.includes("-orange.png")) {
                                currentSrc = currentSrc.replace(".png", "-orange.png");
                            }
                        }
                        iconImg.setAttribute("src", currentSrc);
                    }
                }
            }

            // Click handling directly for vector path wedges
            sectorBgs.forEach(btn => {
                btn.addEventListener("click", () => {
                    const id = btn.getAttribute("data-sector");
                    executeToggle(id);
                });
            });

            // Click handling mirrored for overlapping HTML overlay structures
            overlayBlocks.forEach(block => {
                block.addEventListener("click", () => {
                    const id = block.getAttribute("data-sector");
                    executeToggle(id);
                });
            });

            // Global Master Hub Center Action Rules
            centerHubBtn.addEventListener("click", () => {
                const currentActiveCount = document.querySelectorAll(".wheel-sector-btn.is-active").length;
                const shouldActivateAll = (currentActiveCount === sectorBgs.length) ? false : true;

                sectorBgs.forEach(btn => {
                    const id = btn.getAttribute("data-sector");
                    executeToggle(id, shouldActivateAll);
                });
            });

            // Initialize first view state cleanly with tyre default active (s3)
            executeToggle("s1", false);
            executeToggle("s2", false);
            executeToggle("s3", true);
            executeToggle("s4", false);
            executeToggle("s5", false);
            executeToggle("s6", false);
        });
    </script>
